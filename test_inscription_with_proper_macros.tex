% Inscription macros
\newcommand{\inscriptionref}[1]{\textit{(#1)}}
\newcommand{\inscriptionlabel}[1]{\textbf{(#1:)}}
\newcommand{\inscriptionside}[1]{\emph{#1}}
\newcommand{\inscriptionsection}[1]{%
  \par\vspace{0.5\baselineskip}
  \noindent\hfill\textbf{\MakeUppercase{#1}}\hfill\par
  \noindent\rule{\textwidth}{0.5pt}\par
  \vspace{0.3\baselineskip}
}

\newenvironment{inscriptionpair}
{%
  \par\medskip
  \noindent\begin{tabularx}{\textwidth}{@{}p{1.2cm}XX@{}}
  \hline
}
{%
  \hline
  \end{tabularx}
  \medskip
}

\inscriptionsection{PAIR OF COMPLEMENTARY CHARGES}

\begin{inscriptionpair}
\inscriptionlabel{Preface:} & \inscriptionside{positive, left side} & \inscriptionside{negative, right side} \\
& \inscriptionref{Ping-pien 235.2} & \inscriptionref{Ping-pien 235.1} \\
\hline
\inscriptionlabel{Charge:} & Crack-making on chi-mao (day 16), Ch'üch divined: & Crack-making on chi-mao (day 16), Ch'üch divined: \\
& ``It will rain.'' & ``It will not perhaps rain.'' \\
\hline
\inscriptionlabel{Prognostication:} & The king, reading the cracks (said): & \\
& ``It will rain; it will be a jen day.'' & \\
\hline
\inscriptionlabel{Verification:} & On jen-wu (day 19) it really did rain. & \\
\end{inscriptionpair}
